\subsection{Governed remember}

\begin{lstlisting}[language=Python]
def remember(subject, message, origin):
    with sqlite.begin_immediate():
        episode = store_episode(message, origin)
        audit("episode.ingested", digest(message), origin)
        candidate = deterministic_extract(message, origin)
        if candidate is None:
            return episode

        trust = trust_tier(origin)
        status = active if trust_is_user(trust) else quarantined
        prior = active_same_slot(candidate.fact_key) if status == active else []
        record = insert_record(candidate, episode, trust, status, prior)
        supersede(prior, by=record)  # never runs for quarantined input
        graph_changes = index_or_quarantine(record)
        audit(record_transition(status), record, graph_changes)
        return record
\end{lstlisting}

\subsection{Recall}

\begin{lstlisting}[language=Python]
def recall(subject, query, use_graph=False, limit=10):
    records, fts = active_fts_plus_recent(subject, query, cap=200)
    graph_candidates = {}
    if use_graph:
        seeds = graph_fts(query, cap=16)
        graph_candidates = spread_active_only(
            seeds, depth=2, frontier=64
        )
        records += active_records_nominated_outside_fts(graph_candidates)
    direct = rank(records, text=0.75, trust=0.15, recency=0.10)
    ranked = weighted_rrf(direct, graph_candidates) if use_graph else direct
    audit_bounded_retrieval(query_digest, ranked[:50], returned=ranked[:limit])
    return ranked[:limit]
\end{lstlisting}

\subsection{Guarded commit}

\begin{lstlisting}[language=Python]
def commit(transaction):
    require(transaction.state == APPROVED)
    recompute_and_verify_world_patch(transaction)
    verify_unexpired_exact_plan_approval(transaction)
    verify_adapter_manifests(transaction)
    revalidate_all_preconditions(transaction)
    persist(COMMITTING)

    applied = []
    for operation in dependency_order(transaction):
        if not revalidate(operation):
            return compensate_and_finish(applied)
        persist(operation, EXECUTING)
        try:
            receipt = execute_without_db_transaction(operation)
        except BaseException:
            persist(operation, UNCERTAIN)
            compensate(applied)
            return finish(UNCERTAIN)
        if not verify_postcondition(operation, receipt):
            persist(operation, UNCERTAIN)
            compensate(applied)
            return finish(UNCERTAIN)
        persist(operation, VERIFIED, receipt_digest)
        applied.append(operation)
    return finish(COMMITTED)
\end{lstlisting}

\section{Operational Recovery Notes}
\label{app:recovery}

\subsection{macOS}

Quit the service normally so the current WAL state is checkpointed and sealed. Back up \code{memories.db.enc} and \code{memories.db.enc.hmac} together, plus the assistant workspace. Recovery on another Mac also requires the \code{aetnamem}/\code{database-key} Keychain value. Exporting that value displays a live secret and should occur only into an encrypted recovery process. If graph archive was explicitly enabled, back up its directory separately and encrypt the backup.

\subsection{Linux and Windows}

Stop the service before copying \code{memories.db}, or use a SQLite-aware online backup. Back up the workspace separately. Both database and backup destination are plaintext at the application layer, so use encrypted storage. On Windows, confirm BitLocker status and recovery-key custody. On Linux, confirm the data path is actually inside the intended encrypted volume; installing \code{cryptsetup} alone does not encrypt an existing home directory.
